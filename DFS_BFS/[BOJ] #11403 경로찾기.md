```c++
#include<iostream>
#include<string.h>
#include<vector>
using namespace std;
#define F(x,y,u,p) for(int x = 0; x<u; x++)for(int y = 0; y<p; y++)
int n, arr[105][105];
vector<vector<int>> v;
void dfs(int start, int present) {
	for (auto& e : v[present]) {
		if (arr[start][e]) continue;
		arr[start][e] = 1;
		dfs(start, e);
	}
}
int main() {
	scanf("%d", &n);
	v.resize(n * n);
	int a;
	F(i, j, n, n) {
		scanf("%d", &a);
		if (a) v[i].push_back({ j });
	}
	for (int i = 0; i < n; i++) {
		dfs(i, i);
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			printf("%d ", arr[i][j]);
		}
		printf("\n");
	}
	return 0;
}
```
