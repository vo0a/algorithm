#include <iostream>
using namespace std;
int t, n;
int go(int cnt, int sum) {
	if (sum > n) {
		return 0;
	}
	if (sum == n) {
		return 1;
	}
	int now = 0;
	for (int i = 1; i <= 3; i++) {
		now += go(cnt + 1, sum + i);
	}
	return now;
}
int main() {
	scanf("%d", &t);
	while (t--) {
		scanf("%d", &n);
		printf("%d\n", go(0, 0));
	}
	return 0;
}
