```c++
#include <iostream>
#include <queue>
#include <string.h>
using namespace std;
#define MAX 1000001
int chk[MAX];
int N, st, ed, Up, Down;
int main() {
	scanf("%d%d%d%d%d", &N, &st, &ed, &Up, &Down);
	memset(chk, -1, sizeof(chk));
	queue<int> q;
	q.push(st);
	chk[st] = 0;
	while (!q.empty()) {
		int floor = q.front();
		q.pop();
		if (floor == ed) {
			printf("%d", chk[ed]);
			return 0;
		}

		int next = floor + Up;
		if (next <= N && (chk[next] == -1 || chk[next] > chk[floor] + 1)) {
			chk[next] = chk[floor] + 1;
			q.push(next);
		}
		next = floor - Down;
		if (next > 0 && (chk[next] == -1 || chk[next] > chk[floor] + 1)) {
			chk[next] = chk[floor] + 1;
			q.push(next);
		}		
	}
	
	printf("use the stairs");
	return 0;
}
```
