```c++
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <sstream>
#include <map>
#include <set>
using namespace std;

int n, c;
vector<vector<int>> Player;
vector<vector<string>> Cmd; // 카드번호 - 문자열 or 삭제할 인덱스

vector<string> split(string str, char delimiter) {
	vector<string> ret;
	stringstream ss(str);
	string s;

	while (getline(ss, s, delimiter)) {
		ret.push_back(s);
	}
	return ret;
}

set<string> answer;
string a;
void solve(string p, string s, int cardChk) {
	for (int i = 0; i < p.length(); i++) {
		int player = p[i] - '0';
		for (int j = 0; j < Player[player].size(); j++) {
			int cardNum = Player[player][j];
			if (cardChk & (1 << cardNum)) continue;

			for (int k = 0; k < Cmd[cardNum].size(); k++) {
				string cmd = Cmd[cardNum][k].substr(0, 3);
				string card = Cmd[cardNum][k].substr(4);
				if (cmd == "ADD") {
					s += card;
				}
				else {
					// DEL 안되면 Error 출력하고 return;
					int num = stoi(card);
					if (s.length() == 0 || s.length() <= num) {
						answer.insert("ERROR");
						return;
					}
					s.erase(num, 1);
				}
			}
			cardChk |= (1 << cardNum);
			break;
		}
	}

	if (s.length() == 0) 
		answer.insert("EMPTY");
	else answer.insert(s);
}


int main() {
	ios_base::sync_with_stdio(false); 
	cin.tie(NULL); cout.tie(NULL);
	cin >> n >> c;
	Player.resize(n + 1);
	string permutation = "";
	for (int i = 1, t, num; i <= n; i++) {
		cin >> t;
		while (t--) {
			cin >> num;
			Player[i].push_back(num);
			permutation += to_string(i);
		}
	}

	Cmd.resize(c + 1);
	string s;
	s[0] = getchar();
	for (int i = 1; i <= c; i++) {
		getline(cin, s); // getline(cin, s, '\n');와 같음
		vector<string> sentence = split(s, ',');
		string cmd = "";
		for(auto e : sentence)
			Cmd[i].push_back(e);
	}

	// 순열을 기반으로 시뮬레이션
	do {
		solve(permutation, "", 0);
	} while (next_permutation(permutation.begin(), permutation.end()));

	for (auto e : answer) cout << e << '\n';
	return 0;
}
```
