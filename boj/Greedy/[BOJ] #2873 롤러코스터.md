```c++
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int n, m;
int a[1001][1001];
void append(string& s, char c, int cnt) {
	for (int i = 0; i < cnt; i++) {
		s += c;
	}
}
int main() {
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			scanf("%d", &a[i][j]);
		}
	}
	string s = "";
	if (n % 2 == 1) {
		for (int i = 0; i < n; i++) {
			if (i % 2 == 0) {
				append(s, 'R', m - 1);
				if (i != n - 1) {
					append(s, 'D', 1);
				}
			}
			else {
				append(s, 'L', m - 1);
				append(s, 'D', 1);
			}
		}
	}
	else if (m % 2 == 1) {
		for (int j = 0; j < m; j++) {
			if (j % 2 == 0) {
				append(s, 'D', n - 1);
				if (j != m - 1) {
					append(s, 'R', 1);
				}
			}
			else {
				append(s, 'U', n - 1);
				append(s, 'R', 1);
			}
		}
	}
	else { // n, m 이 모두 짝수인 경우 한 칸만 방문 못함
		int x, y;
		x = 0;
		y = 1;
		// 가장 크기가 작은 칸을 방문하지 않기 위해 최소값을 찾음
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if ((i + j) % 2 == 1) {
					if (a[x][y] > a[i][j]) {
						x = i;
						y = j;
					}
				}
			}
		}
		int x1 = 0;
		int y1 = 0;
		int x2 = n - 1;
		int y2 = m - 1;
		string s2 = "";
		// 두 칸씩 줄여감
		while (x2 - x1 > 1) {
			if (x1 / 2 < x / 2) {
				append(s, 'R', m - 1);
				append(s, 'D', 1);
				append(s, 'L', m - 1);
				append(s, 'D', 1);
				x1 += 2;
			}
			if (x / 2 < x2 / 2) {
				append(s2, 'R', m - 1);
				append(s2, 'D', 1);
				append(s2, 'L', m - 1);
				append(s2, 'D', 1);
				x2 -= 2;
			}
		}
		while (y2 - y1 > 1) {
			if (y1 / 2 < y / 2) {
				append(s, 'D', 1);
				append(s, 'R', 1);
				append(s, 'U', 1);
				append(s, 'R', 1);
				y1 += 2;
			}
			if (y / 2 < y2 / 2) {
				append(s2, 'D', 1);
				append(s2, 'R', 1);
				append(s2, 'U', 1);
				append(s2, 'R', 1);
				y2 -= 2;
			}
		}
		// 항상 2*2 배열이 남음
		if (y == y1) {
			append(s, 'R', 1);
			append(s, 'D', 1);
		}
		else {
			append(s, 'D', 1);
			append(s, 'R', 1);
		}
		reverse(s2.begin(), s2.end());
		s += s2;
	}
	cout << s;
	return 0;
}
```

- 4방향으로 움직일 수 있기 때문에 다이나믹 문제가 아님
- R또는 C가 홀수면 모든 칸을 방문할 수 있지만 모두 짝수면 모든 칸을 방문하는 것은 불가능하고 딱 한 칸만 제외하면 모든 칸을 방문할 수 있다.
