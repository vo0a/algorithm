## 1257. [S/W 문제해결 응용] 6일차 - K번째 문자열

[문제](https://swexpertacademy.com/main/code/problem/problemDetail.do?contestProbId=AV18KWf6ItECFAZN)



```c++
#include <iostream>
#include <algorithm>
#include <map>
#include <string>
 
using namespace std;
 
map<string, int> mp;
 
int main() {
    int tCase;
    scanf("%d", &tCase);
 
    for (int tc = 1; tc <= tCase; tc++) {
        int k;
        scanf("%d", &k);
        mp.clear();
 
        string str;
        char tmp[402];
        scanf("%s", tmp);
        str = tmp;
 
        int len = str.size();
        for (int i = 0; i < len; i++)
            for (int j = 1; j <= len - i; j++)
                mp[str.substr(i, j)] = 1;
 
        bool chk = false;
        for (auto it = mp.begin(); it != mp.end(); it++, k--) {
            if (k == 1) {
                printf("#%d %s\n", tc, it->first.c_str());
                chk = true;
                break;
            }
        }
 
        if (!chk)
            printf("#%d none\n", tc);
    }
 
    return 0;
}
```
