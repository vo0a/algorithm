## 146. LRU Cache

> Medium

[문제](https://leetcode.com/problems/lru-cache/)



```sql
class LRUCache {
public:
    map<int, int> KeyToTime; // 
    map<int, int> TimeToKey; // Time 작은 순서대로 바로 알 수 있게
    map<int, int> cache;
    int C;
    int Time;
    
    LRUCache(int capacity) {
        C = capacity;
        Time = 0;
    }
    
    int get(int key) {
        if(cache.count(key)) {            
            int time = KeyToTime[key];
            TimeToKey.erase(time);
            
            KeyToTime[key] = Time;
            TimeToKey[Time++] = key;
            
            return cache[key];
        }
        else return -1;
    }
    
    void put(int key, int value) {
        if(cache.count(key)) {
                int time = KeyToTime[key];
                TimeToKey.erase(time);

                cache[key] = value;
                KeyToTime[key] = Time;
                TimeToKey[Time++] = key;
                return;
        }
        
        if(cache.size() >= C) {       
            // 삭제
            auto removeIt = TimeToKey.begin(); // 가장 시간이 작은것 선택
            int prevTime = removeIt->first;
            int removekey = removeIt->second;            
            TimeToKey.erase(prevTime);
            KeyToTime.erase(removekey);
            cache.erase(removekey);
            
            // 삽입
            cache[key] = value;
            KeyToTime[key] = Time;
            TimeToKey[Time++] = key;
        }
        else { 
            cache[key] = value;
            KeyToTime[key] = Time;
            TimeToKey[Time++] = key;            
        }
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
```
