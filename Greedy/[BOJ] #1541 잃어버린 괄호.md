#include<cstdio>
#include<iostream>
#include <string>
using namespace std;

char arr[51], sign;
int ans;
string s, num;

int main() {
	cin >> s;

	sign = '+';
	for (int i = 0; i < s.length(); i++) {
		if (s[i] >= '0' && s[i] <= '9') {
			num += s[i];
		}
		else {
			if (sign == '-') {
				ans -= stoi(num);
				num = "";					
			}
			else {
				ans += stoi(num);
				num = "";
				if (s[i] == '-')
					sign = '-';
			}
		}
	}

	if (num != "") {
		if (sign == '+') {
			ans += stoi(num);
		}
		else {
			ans -= stoi(num);
		}
	}
	printf("%d", ans);

	return 0;
}
