## [BOJ] #14891 톱니바퀴

| 시간 제한 | 메모리 제한 | 정답 비율 |

입력이 띄어쓰기 없이 주어지는데 그 부분을 나는 문자열로 받았다가 int로 변경했는데, 더 쉬운 방법이 있었다. 분명히 알았는데 까먹었던 방법 ... 형식지정자 %1d 를 사용하면 띄어쓰기 없이 주어진 input도 하나씩 인식하게 된다.

```c++
for (int i = 1; i <= 4; i++) {
		for (int j = 0; j < 8; j++) {
			scanf("%1d", &gears[i][j]);
		}
}
```



양쪽 끝 부분도 돌아가게 하기 위하여 -1로 초기화 시켜두고 맞물린 톱니의 극이 같으면 회전하고 같지 않다면 반복문을 벗어난다.



| 메모리  | 시간 |
| ------- | ---- |
| 1116 KB | 0 ms |

```c++
#include <cstdio>
#include <string.h>
#include <algorithm>
using namespace std;
int gears[5][10], idx[5], t, gear, dir, preidx, predir, ans; 
char c[10];
int main(){
	memset(gears, -1, sizeof(gears));
	for (int i = 0; i < 4; i++) {
		scanf("%s", c);
		for (int j = 0; j < 8; j++) {
			gears[i][j] = c[j] - '0';
		}
	}
	for (int i = 0; i < 4; i++) idx[i] = 0; // 12시 방향 인덱스는 0으로 초기화.

	scanf("%d", &t);
	while (t--) {
		scanf("%d %d", &gear, &dir);
		// 같지 않으면 돌아감.
		if (gear > 1 && gears[gear - 1][(idx[gear - 1] + 6) % 8] != gears[gear - 2][(idx[gear - 2] + 2) % 8]) {
			predir = dir;
			for (int i = gear - 2; i >= 0; i--) {
				preidx = idx[i];
				if (predir == 1) { // 이전이 시계방향이었다면 반시계방향으로 돌아야함.
					idx[i] = (idx[i] + 1) % 8;
					predir = -1;
				}
				else { // 이전이 반시계방향이었다면 시계방향으로 돌아야함.
					idx[i] = (idx[i] + 7) % 8;
					predir = 1;
				}
				// preidx 와 왼쪽 톱니바퀴를 비교.
				if (gears[i][(preidx + 6) % 8] == gears[i - 1][(idx[i - 1] + 2) % 8]) break;
			}
		}
		if (gear < 4 && gears[gear - 1][(idx[gear - 1] + 2) % 8] != gears[gear][(idx[gear] + 6) % 8]) {
			predir = dir;
			for (int i = gear; i <= 3; i++) {
				preidx = idx[i];
				if (predir == 1) { // 이전이 시계방향이었다면 반시계방향으로 돌아야함.
					idx[i] = (idx[i] + 1) % 8;
					predir = -1;
				}
				else { // 이전이 반시계방향이었다면 시계방향으로 돌아야함.
					idx[i] = (idx[i] + 7) % 8;
					predir = 1;
				}
				// preidx 와 오른쪽 톱니바퀴를 비교.
				if (gears[i][(preidx + 2) % 8] == gears[i + 1][(idx[i + 1] + 6) % 8]) break; // 양쪽 끝 부분도 돌아가게 하기 위하여 -1로 초기화 시켜두고 같은지로 판별.
			}
		}
		// gear -1 번째 톱니 idx 변경.
		if (dir == 1) { // 시계방향
			idx[gear - 1] = (idx[gear - 1] + 7) % 8;
		}
		else { // 반시계방향
			idx[gear - 1] = (idx[gear - 1] + 1) % 8;
		}
	}
	int sq = 1;
	for (int i = 0; i < 4; i++) { // S극 은 1로 나타나있고 
		ans += gears[i][idx[i]] * sq;	sq *= 2;
	}
	printf("%d", ans);

	return 0;
}
```
